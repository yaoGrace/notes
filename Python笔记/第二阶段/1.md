"""
其他内置方法，学习常用的几个
"""


class Student:
    def __init__(self, name, age):
        self.name = name  # 定义学生姓名属性，并赋值
        self.age = age  # 定义学生年龄属性，并赋值

    # __str__ 魔术方法 字符串方法
    def __str__(self):
        return f"Student类对象，name:{self.name} , age:{self.age}"

    # __lt__ 魔术方法 小于。大于符号比较
    def __lt__(self, other):
        return self.age < other.age

    # __le__ 魔术方法 小于等于、大于等于符号比较
    def __le__(self, other):
        return self.age <= other.age

    # __eq__ 魔术方法 == 符号比较
    def __eq__(self, other):
        return self.age == other.age


stu1 = Student("周杰伦", 31)
stu2 = Student("林俊杰", 36)

print(stu1)
print(str(stu1))

print(stu1 < stu2)
print(stu1 > stu2)

print(stu1 <= stu2)
print(stu1 >= stu2)

print(stu1 == stu2)
```

"""
    演示面向对象封装思想中私有成员的使用
"""


# 定义一个类，内含私有成员变量和私有成员方法

class Phone:
    __current_voltage = 0  # 当前手机运行电压 私有成员变量

    def __keep_single_core(self):
        """
        私有成员方法
        :return:
        """
        print("让CPU以单核模式运行")

    def call_by_5g(self):
        if self.__current_voltage >= 1:
            print("5g通话已开启")
        else:
            # 电量<1 运行单核模式
            self.__keep_single_core()
            print("电量不足，无法使用5g通话，并已设置为单核运行进行省电。")


phone = Phone()

# 私有成员方法无法直接调用，此处报错
# phone.__keep_single_core()

# 私有成员变量 也无法直接使用，此处报错
# print(phone.__current_voltage)


phone.call_by_5g()

"""
练习题：设计带有私有成员的手机
"""


class Phone:
    # 提供私有成员变量： __is_5g_enable
    __is_5g_enable = False  # 5g状态

    # 提供私有成员方法 ：__check_5g()
    def __check_5g(self):
        if self.__is_5g_enable:
            print("5g开启")
        else:
            print("5g关闭，使用4g网络")

    # 提供公开成员方法：call_by_5g()
    def call_by_5g(self):
        # 判断5g网络状态
        self.__check_5g()
        print("正在通话中.....")


# 调用
phone = Phone()
phone.call_by_5g()
```
